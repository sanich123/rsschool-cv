## Example of code:
```
var makeFancyString = function(s) {
    let result = '';
    let count = 0;
    for (let i = 0; i < s.length; i++) {
        if (s[i] === s[i - 1]) {
            count++;
            if (count <= 2) {
                result += s[i];
            }
        } else {
            count = 1;
            result += s[i];
        }
    } 

    return result;
};
```
